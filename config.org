(defun julina/dired-open-smartly()
  "Open PDFs in Zathura, videos in VLC, and everything else in Emacs."
  (interactive)
  (let ((file (dired-get-file-for-visit)))
    (cond
     ;; If it's a PDF, launch Zathura asynchronously
     ((string-match-p "\\.pdf\\'" file)
      (start-process "zathura-process" nil "zathura" file))

     ;; If it's a video, launch VLC asynchronously
     ((string-match-p "\\.\\(mp4\\|mkv\\|webm\\|avi\\)\\'" file)
      (start-process "vlc-process" nil "vlc" file))

     ;; Otherwise, open it normally inside Emacs
     (t
      (find-file file)))))
(map! :leader
      (:prefix ("d" . "dired")
       :desc "Open dired" "d" #'dired
       :desc "Dired jump to current" "j" #'dired-jump)
      (:after dired
              (:map dired-mode-map
               :desc "Peep-dired image previews" "d p" #'peep-dired
               :desc "Dired view file"           "d v" #'dired-view-file)))

;; Wrap all Dired config so it only evaluates AFTER Dired is loaded
(after! dired
  ;; Force our custom function to take over 'l' and 'Enter' in Normal mode
  (map! :map dired-mode-map
        :n "l"     #'julina/dired-open-smartly
        :n "RET"   #'julina/dired-open-smartly

        TeX-view-program-selection '((output-pdf "Zathura")))

   TeX-source-correlate-mode t ; enable forward/backward search
   TeX-auto-save t ; parse on save
   TeX-parse-self t ; parse on load
   TeX-save-query nil
   TeX-PDF-mode t ; compile to pdf by default
   TeX-command-default "LatexMk" ; latexmk for compilation
   )
  (setq-default TeX-master nil)

  ;; lsp
  (setq
   lsp-tex-server 'digestif)

  ;; hooks
  (add-hook 'LaTeX-mode-hook #'lsp)
  (add-hook 'LaTeX-mode-hook #'corfu-mode)
  (add-hook 'LaTeX-mode-hook #'turn-on-cdlatex)

  (add-hook! '(LaTeX-mode-hook latex-mode-hook))
  (defun julina/kill-lsp-format ()
    ;; turn off formatting by lsp (digestif doesn't support it)
    (setq-local +format-with-lsp nil)
    (setq-local lsp-enable-formatting nil))

  (defun julina/latex-after-save ()
    "Run latexindent, safely reload, clean backups, and compile quietly."
    (when (member major-mode '(latex-mode LaTeX-mode))
      (let ((file (buffer-file-name)))
        (when file
          (message "Running latexindent...")
          (call-process "latexindent" nil nil nil "-w" file)

          (mapc #'delete-file
                (directory-files (file-name-directory file) t "\\.bak[0-9]*\\'"))

          (revert-buffer t t t)

          (message "Formatted perfectly! Compiling PDF...")
          (save-some-buffers t)
          (TeX-command-run-all nil)))))

  (add-hook 'after-save-hook #'julina/latex-after-save)
  (add-hook 'LaTeX-mode-hook #'julina/kill-lsp-format))
